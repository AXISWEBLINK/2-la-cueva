-- Opcional: normalizar grupos por defecto cuando el proveedor tiene un solo grupo activo.
UPDATE price_provider_groups pg
INNER JOIN (
  SELECT proveedor_id, MIN(id) AS group_id
  FROM price_provider_groups
  WHERE is_active = 1
  GROUP BY proveedor_id
  HAVING COUNT(*) = 1
) single_group
  ON single_group.group_id = pg.id
SET pg.is_default = 1
WHERE pg.is_default = 0;
